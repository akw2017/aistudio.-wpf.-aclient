using AIStudio.Wpf.BasePage.ViewModels;
using AIStudio.Wpf.Business.DTOModels;
using AIStudio.Wpf.Business;
using Prism.Ioc;
using Prism.Mvvm;
using System;
using System.Collections.Generic;
using System.Collections.ObjectModel;
using System.Linq;

namespace AIStudio.Wpf.%areaName%.ViewModels
{
    public class %entityName%EditViewModel : BaseEditViewModel<%entityName%DTO>
    {
        public %entityName%EditViewModel(%entityName%DTO data, string area, string title = "编辑表单") : base(data, area, title)
        {
            if (Data == null)
            {
                InitData();
            }
            else
            {
                GetData(Data);
            }
        }

		protected override void InitData()
		{
			Data = new %entityName%DTO();
		}

        protected override void GetData(%entityName%DTO para)
        {
            base.GetData(para);
        }       
    }
}
